# Тестовое задание для "Траектория"

Scheduler - класс для работы с расписанием сотрудника. Позволяет находить свободные и занятые промежутки времени в графике.

## Пример
```python
from scheduler import Scheduler


scheduler = Scheduler(url="https://ofc-test-01.spb.su/test-task/")
# Данные из url
"""
    "days": [
            {"id":1,"date":"2025-02-15","start":"09:00","end":"21:00"},
            {"id":2,"date":"2025-02-16","start":"08:00","end":"22:00"},
            {"id":3,"date":"2025-02-17","start":"09:00","end":"18:00"},
            {"id":4,"date":"2025-02-18","start":"10:00","end":"18:00"},
            {"id":5,"date":"2025-02-19","start":"09:00","end":"18:00"}
    ],
    "timeslots":[
            {"id":1,"day_id":1,"start":"17:30","end":"20:00"},
            {"id":2,"day_id":1,"start":"09:00","end":"12:00"},
            {"id":3,"day_id":2,"start":"14:30","end":"18:00"},
            {"id":4,"day_id":2,"start":"09:30","end":"11:00"},
            {"id":5,"day_id":3,"start":"12:30","end":"18:00"},
            {"id":6,"day_id":4,"start":"10:00","end":"11:00"},
            {"id":7,"day_id":4,"start":"11:30","end":"14:00"},
            {"id":8,"day_id":4,"start":"14:00","end":"16:00"},
            {"id":9,"day_id":4,"start":"17:00","end":"18:00"}
    ]
"""

# Получение занятых слотов
busy = scheduler.get_busy_slots("2025-02-15")
print("Занятые интервалы:", busy)
# Занятые интервалы: [('10:00', '12:00'), ('14:00', '15:30')]

# Получение свободных слотов
free = scheduler.get_free_slots("2025-02-15")
print("Свободные интервалы:", free)
# Свободные интервалы: [('09:00', '10:00'), ('12:00', '14:00'), ('15:30', '21:00')]

# Проверка доступности конкретного времени
is_free = scheduler.is_available("2025-02-15", "12:30", "13:30")
print("Свободен ли интервал 12:30–13:30?", is_free)
# Свободен ли интервал 12:30–13:30? True

# Поиск первого подходящего свободного слота для записи (в минутах)
slot = scheduler.find_slot_for_duration(60)
print("Первый свободный слот на 60 минут:", slot)
# Первый свободный слот на 60 минут: ('2025-02-15', '09:00', '10:00')
```
